const API_KEY = '$2a$10$EX9QUVZw3nVG3ndTgiBfqO7dP8EAIDw4II0a58ktftvwfEc4yJsvW';

const BIN_ID  = '6a3554cfda38895dfede0b4b';

// ═══════════════════════════════════════════════════════════


const BIN_URL = `https://api.jsonbin.io/v3/b/${BIN_ID}`;

const HDR     = { 'X-Master-Key': API_KEY, 'Content-Type': 'application/json' };


let S          = null;   // estado global

let toastTimer = null;


// ── Utilidades ────────────────────────────────────────────

function uid()  { return Math.random().toString(36).slice(2, 10); }

function esc(v) { return String(v).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;'); }

function today() {

  const d = new Date();

  return `${String(d.getDate()).padStart(2,'0')}/${String(d.getMonth()+1).padStart(2,'0')}`;

}


function toast(msg, err = false) {

  const el = document.getElementById('toast');

  el.textContent = msg;

  el.className   = 'show' + (err ? ' err' : '');

  clearTimeout(toastTimer);

  toastTimer = setTimeout(() => el.className = '', 2800);

}


function openSheet(id)  { document.getElementById(id).classList.add('open'); }

function closeSheet(id) { document.getElementById(id).classList.remove('open'); }


// ── JSONBin API ───────────────────────────────────────────

async function load() {

  const r = await fetch(`${BIN_URL}/latest`, { headers: HDR });

  if (!r.ok) throw new Error(`HTTP ${r.status}`);

  const d = await r.json();

  S = normalize(d.record);

}


async function save() {

  const r = await fetch(BIN_URL, { method: 'PUT', headers: HDR, body: JSON.stringify(S) });

  if (!r.ok) throw new Error(`HTTP ${r.status}`);

}


async function persist(msg, closeId) {

  try {

    await save();

    toast(msg);

    if (closeId) closeSheet(closeId);

    render();

  } catch(e) {

    toast('Error al guardar: ' + e.message, true);

  }

}


// ── Normalización de estado ───────────────────────────────

function normalize(raw) {

  return {

    productNames: {

      A: raw?.productNames?.A || 'Producto A',

      B: raw?.productNames?.B || 'Producto B'

    },

    members: (raw?.members || []).map(m => ({

      id:       m.id       || uid(),

      name:     m.name     || '',

      status:   m.status   || 'pending',

      lastDate: m.lastDate || null

    })),

    assignments: {

      A: raw?.assignments?.A || null,

      B: raw?.assignments?.B || null

    },

    roundNumber: raw?.roundNumber || 1

  };

}


// ── Computed ──────────────────────────────────────────────

function assignedProducts(id) {

  const a = S.assignments.A === id;

  const b = S.assignments.B === id;

  if (a && b) return 'AB';

  if (a)      return 'A';

  if (b)      return 'B';

  return null;

}


function completedCount() {

  return S.members.filter(m => m.status === 'completed').length;

}


// ── Acciones de estado ────────────────────────────────────

async function doAssign(newA, newB) {

  // Miembros que pierden su asignación actual vuelven a 'pending'

  const prev = new Set([S.assignments.A, S.assignments.B].filter(Boolean));

  const next  = new Set([newA, newB].filter(Boolean));

  prev.forEach(id => {

    if (!next.has(id)) {

      const m = S.members.find(x => x.id === id);

      if (m && m.status === 'assigned') m.status = 'pending';

    }

  });

  // Aplicar nuevas asignaciones

  S.assignments.A = newA || null;

  S.assignments.B = newB || null;

  next.forEach(id => {

    const m = S.members.find(x => x.id === id);

    if (m) m.status = 'assigned';  // permite reasignar aunque ya completó

  });

  await persist('Turno asignado ✓', 'm-assign');

}


async function doConfirm(id) {

  const m = S.members.find(x => x.id === id);

  if (!m) return;

  m.status   = 'completed';

  m.lastDate = today();

  if (S.assignments.A === id) S.assignments.A = null;

  if (S.assignments.B === id) S.assignments.B = null;

  await persist(`${m.name} marcado como entregado ✓`);

}


async function doReset() {

  S.members.forEach(m => m.status = 'pending');

  S.assignments = { A: null, B: null };

  S.roundNumber = (S.roundNumber || 1) + 1;

  await persist('¡Nueva ronda iniciada! ✓', 'm-settings');

}


async function doSaveMembers(members) {

  S.members = members;

  if (S.assignments.A && !members.find(m => m.id === S.assignments.A)) S.assignments.A = null;

  if (S.assignments.B && !members.find(m => m.id === S.assignments.B)) S.assignments.B = null;

  await persist('Miembros actualizados ✓', 'm-members');

}


async function doSaveSettings(nameA, nameB) {

  S.productNames.A = nameA || 'Producto A';

  S.productNames.B = nameB || 'Producto B';

  await persist('Ajustes guardados ✓', 'm-settings');

}


// ── Handlers de UI ────────────────────────────────────────

function handleAssign() {

  const a = document.getElementById('sel-a').value;

  const b = document.getElementById('sel-b').value;

  if (!a && !b) { toast('Selecciona al menos un miembro', true); return; }

  doAssign(a || null, b || null);

}


function handleConfirm(id) {

  const m = S.members.find(x => x.id === id);

  if (!m) return;

  if (!confirm(`¿Confirmar que ${m.name} ha realizado la entrega?`)) return;

  doConfirm(id);

}


function handleReset() {

  const total = S.members.length;

  const done  = completedCount();

  const msg   = done < total

    ? `Aún quedan ${total - done} miembros sin completar. ¿Iniciar nueva ronda de todas formas?`

    : '¿Iniciar una nueva ronda? Todos los miembros volverán a "pendiente".';

  if (!confirm(msg)) return;

  doReset();

}


function handleSaveMembers() {

  const rows    = document.querySelectorAll('#edit-list .edit-row');

  const members = [];

  let valid     = true;

  rows.forEach(row => {

    const input = row.querySelector('input');

    const name  = input.value.trim();

    if (!name) { valid = false; input.classList.add('error'); return; }

    input.classList.remove('error');

    const id       = row.dataset.id;

    const existing = S.members.find(m => m.id === id);

    members.push(existing ? { ...existing, name } : { id, name, status: 'pending', lastDate: null });

  });

  if (!valid) { toast('Hay campos en blanco', true); return; }

  doSaveMembers(members);

}


function handleSaveSettings() {

  const a = document.getElementById('set-a').value.trim();

  const b = document.getElementById('set-b').value.trim();

  doSaveSettings(a, b);

}


// ── Apertura de modales ───────────────────────────────────

function openAssign() {

  document.getElementById('lbl-a').textContent = S.productNames.A;

  document.getElementById('lbl-b').textContent = S.productNames.B;


  const buildOpts = (currentVal) => [

    '<option value="">— Sin asignar —</option>',

    ...S.members.map(m => {

      const star = m.status === 'completed' ? ' ★' : '';

      const sel  = currentVal === m.id ? ' selected' : '';

      return `<option value="${esc(m.id)}"${sel}>${esc(m.name)}${star}</option>`;

    })

  ].join('');


  document.getElementById('sel-a').innerHTML = buildOpts(S.assignments.A);

  document.getElementById('sel-b').innerHTML = buildOpts(S.assignments.B);

  openSheet('m-assign');

}


function openMembers() {

  document.getElementById('edit-list').innerHTML = S.members.map(m => `

    <div class="edit-row" data-id="${esc(m.id)}">

      <input type="text" value="${esc(m.name)}" placeholder="Nombre del miembro">

      <button class="btn-icon" onclick="this.closest('.edit-row').remove()" title="Eliminar">🗑</button>

    </div>

  `).join('');

  openSheet('m-members');

}


function addEditRow() {

  const list = document.getElementById('edit-list');

  const div  = document.createElement('div');

  div.className  = 'edit-row';

  div.dataset.id = uid();

  div.innerHTML  = `

    <input type="text" placeholder="Nombre del miembro">

    <button class="btn-icon" onclick="this.closest('.edit-row').remove()" title="Eliminar">🗑</button>

  `;

  list.appendChild(div);

  div.querySelector('input').focus();

}


function openSettings() {

  document.getElementById('set-a').value = S.productNames.A;

  document.getElementById('set-b').value = S.productNames.B;

  openSheet('m-settings');

}


async function refreshData() {

  try {

    await load();

    render();

    toast('Datos actualizados ✓');

  } catch(e) {

    toast('Error al actualizar', true);

  }

}


// ── Render principal ──────────────────────────────────────

function render() {

  const app = document.getElementById('app');


  // Estado de error (sin conexión o sin configurar)

  if (!S) {

    app.innerHTML = `

      <div class="error-card">

        <div class="eicon">⚠️</div>

        <h2>No se pudo conectar</h2>

        <p>Comprueba que <strong>API_KEY</strong> y <strong>BIN_ID</strong> están configurados correctamente en el archivo HTML y que el Bin existe en JSONBin.</p>

      </div>`;

    return;

  }


  const total    = S.members.length;

  const done     = completedCount();

  const pct      = total > 0 ? Math.round(done / total * 100) : 0;

  const allDone  = total > 0 && done === total;

  const isSetup  = API_KEY !== 'TU_API_KEY_AQUI' && BIN_ID !== 'TU_BIN_ID_AQUI';


  // Aviso de configuración pendiente

  const configWarn = isSetup ? '' : `

    <div class="config-warn">

      <b>⚠️ App sin configurar</b>

      Edita el archivo HTML y sustituye <code>TU_API_KEY_AQUI</code> y <code>TU_BIN_ID_AQUI</code> por los valores de JSONBin.

    </div>`;


  // Sección de turno actual

  const assignedIds = [...new Set([S.assignments.A, S.assignments.B].filter(Boolean))];

  let assignSection = '';


  if (assignedIds.length > 0) {

    const cards = assignedIds.map(id => {

      const m = S.members.find(x => x.id === id);

      if (!m) return '';

      const ap       = assignedProducts(id);

      const isDone   = m.status === 'completed';

      const prodName = ap === 'AB'

        ? `${S.productNames.A} + ${S.productNames.B}`

        : (ap === 'A' ? S.productNames.A : S.productNames.B);

      const chipLabel = ap; // 'A', 'B', o 'AB'


      return `

        <div class="assign-card${isDone ? ' done' : ''}">

          <div class="product-chip">${esc(chipLabel)}</div>

          <div class="assign-info">

            <div class="assign-name">${esc(m.name)}</div>

            <div class="assign-sub">${esc(prodName)}</div>

          </div>

          ${isDone

            ? `<div class="chip-done">✓ Entregado</div>`

            : `<button class="btn-confirm" onclick="handleConfirm('${id}')">Confirmar</button>`

          }

        </div>`;

    }).join('');


    assignSection = `

      <div class="section-label">Turno actual</div>

      ${cards}

      <div class="mb-12"></div>`;

  }


  // Lista de miembros

  let memberList = '';

  if (total === 0) {

    memberList = `

      <div class="empty">

        <div class="empty-icon">👥</div>

        <div class="empty-title">Sin miembros todavía</div>

        <p class="txt-m">Pulsa "Miembros" en la esquina superior para añadir personas</p>

      </div>`;

  } else {

    const rows = S.members.map(m => {

      const ap = assignedProducts(m.id);

      const st = m.status || 'pending';

      let tags = '';

      if (ap === 'A' || ap === 'AB') tags += `<span class="tag tag-a">${esc(S.productNames.A)}</span>`;

      if (ap === 'B' || ap === 'AB') tags += `<span class="tag tag-b">${esc(S.productNames.B)}</span>`;

      if (m.lastDate)                tags += `<span class="tag tag-date">📅 ${esc(m.lastDate)}</span>`;


      return `

        <div class="member-row ${st}">

          <div class="member-info">

            <div class="member-name">${esc(m.name)}</div>

            ${tags ? `<div class="member-tags">${tags}</div>` : ''}

          </div>

        </div>`;

    }).join('');

    memberList = `<div class="member-list">${rows}</div>`;

  }


  // Renderizado completo

  app.innerHTML = `

    ${configWarn}


    <div class="app-header">

      <div>

        <div class="app-title">Turno de Clase</div>

        <div class="app-meta">

          <span class="round-pill">Ronda ${S.roundNumber}</span>

          <span class="txt-m">${done} de ${total} completados</span>

        </div>

      </div>

      <div class="header-actions">

        <button class="btn-outline" onclick="openMembers()">👥 Miembros</button>

        <button class="btn-outline" onclick="openSettings()">⚙</button>

        <button class="btn-outline" onclick="refreshData()" title="Actualizar datos">↻</button>

      </div>

    </div>


    <div class="progress-card mb-12">

      <div class="progress-inner">

        <div class="progress-frac">${done}<sup>/${total}</sup></div>

        <div class="progress-meta">

          <div class="progress-text">${allDone ? '🎉 ¡Todos han completado la ronda!' : `Progreso de la ronda · ${pct}%`}</div>

          <div class="progress-track">

            <div class="progress-fill" style="width:${pct}%"></div>

          </div>

        </div>

      </div>

    </div>


    ${assignSection}


    <div class="action-bar">

      <button class="btn-primary" onclick="openAssign()" ${total === 0 ? 'disabled' : ''}>

        📋 Asignar turno

      </button>

      ${allDone ? `<button class="btn-outline" onclick="handleReset()">🔄 Nueva ronda</button>` : ''}

    </div>


    <div class="section-label">Lista de miembros</div>

    ${memberList}

  `;

}

      <span class="txt-m">Cargando datos...</span>

    </div>`;

  try { await load(); } catch(e) { S = null; }

  render();

}
